## Core instance methods

<MethodSignature
  name="init"
  signature={`Future<void> init({
  required String apiKey,
  required String endpoint,
  bool debug = false,
  bool trackLifecycleEvents = true,
  bool enableSessionReplay = false,
  SankofaReplayMode replayMode = SankofaReplayMode.wireframe,
})`}
  description="Initializes the SDK, resolves endpoint URLs, loads persisted identity and queue state, gathers default properties, and prepares replay when enabled."
/>

<MethodSignature
  name="track"
  signature={`Future<void> track(
  String eventName, [
  Map<String, dynamic>? properties,
])`}
  description="Refreshes the current session, serializes event properties, injects $session_id, adds default properties, and appends the event to the local queue."
/>

<MethodSignature
  name="identify"
  signature={`Future<void> identify(String userId)`}
  description="Switches the active distinct ID to a known user and queues an alias item when the previous identity differs."
/>

<MethodSignature
  name="peopleSet"
  signature={`Future<void> peopleSet(Map<String, dynamic> properties)`}
  description="Queues a people profile update for the active distinct ID using the transport serializer."
/>

<MethodSignature
  name="reset"
  signature={`Future<void> reset()`}
  description="Flushes pending events, clears the current user ID, creates a new anonymous ID, rotates the session, and reconfigures replay if enabled."
/>

<MethodSignature
  name="resolveTrackUri"
